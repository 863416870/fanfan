    QueryWrapper<User> queryWrapper = new QueryWrapper<User>();

     /** if(!StringUtils.isEmpty(name)){
            queryWrapper.like("username",name);
        }

            queryWrapper.like("mobile",mobile);
        }
        **/

                .like(!StringUtils.isEmpty(name),"username",name)

}
